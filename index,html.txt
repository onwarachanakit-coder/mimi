     {

       options: [

         "Musta ja kirkkaan oranssi."
       ],
       correctIndex: 1
     },
     {
       category: "arts",
       text: "Kun sävellät pientä melodiaa, mikä tekee siitä helpommin muistettavan?",
       options: [
         "Lyhyt, toistuva teema, jossa on selkeä rytmi.",
         "Hyvin pitkä ja monimutkainen sävelkulku.",
         "Satunnaiset äänet ilman mitään toistoa."
       ],
       correctIndex: 0
     },
     // MATH SKILLS
     {
       category: "math",
       text: "Jos sinulla on 3 × 4 omenaa, montako omenaa sinulla on yhteensä?",
       options: [
         "7",

       ],
       correctIndex: 1
     },
     {
       category: "math",
       text: "Luku 0,25 on sama kuin...",
       options: [

         "1/3",
         "1/4"
       ],
       correctIndex: 2
     },
     {
       category: "math",
       text: "Jos bussilippu maksaa 2,40 € ja sinulla on 10 €, paljonko rahaa jää jäljelle?",
       options: [
         "7,60 €",
         "6,40 €",
         "8,40 €"
       ],
       correctIndex: 0
     }
   ];
   // -----------------------------
   // STATE
   // -----------------------------
   let currentIndex = 0;
   let totalScore = 0;
   const categoryScores = {
     social: 0,
     arts: 0,
     math: 0
   };
   const categoryTotals = {
     social: questions.filter(q => q.category === "social").length,
     arts: questions.filter(q => q.category === "arts").length,
     math: questions.filter(q => q.category === "math").length
   };
   // -----------------------------
   // ELEMENTS
   // -----------------------------
   const startScreen = document.getElementById("start-screen");
   const quizScreen = document.getElementById("quiz-screen");
   const resultScreen = document.getElementById("result-screen");
   const startBtn = document.getElementById("start-btn");
   const nextBtn = document.getElementById("next-btn");
   const restartBtn = document.getElementById("restart-btn");
   const categoryTag = document.getElementById("category-tag");
   const progressText = document.getElementById("progress-text");
   const questionText = document.getElementById("question-text");
   const optionsContainer = document.getElementById("options-container");
   const scoreTotalEl = document.getElementById("score-total");
   const socialScoreEl = document.getElementById("social-score");
   const artsScoreEl = document.getElementById("arts-score");
   const mathScoreEl = document.getElementById("math-score");
   const socialCommentEl = document.getElementById("social-comment");
   const artsCommentEl = document.getElementById("arts-comment");
   const mathCommentEl = document.getElementById("math-comment");
   const overallScoreEl = document.getElementById("overall-score");
   // -----------------------------
   // HELPERS
   // -----------------------------
   function showScreen(screen) {
     [startScreen, quizScreen, resultScreen].forEach(s => s.classList.remove("active"));
     screen.classList.add("active");
   }
   function getCategoryLabel(cat) {
     if (cat === "social") return "Social skills";
     if (cat === "arts") return "Arts skills";
     if (cat === "math") return "Math skills";
     return cat;
   }
   function setCategoryTag(cat) {
     categoryTag.className = "category-tag";
     if (cat === "social") categoryTag.classList.add("category-social");
     if (cat === "arts") categoryTag.classList.add("category-arts");
     if (cat === "math") categoryTag.classList.add("category-math");
     categoryTag.textContent = getCategoryLabel(cat);
   }
   function renderQuestion() {
     const q = questions[currentIndex];
     setCategoryTag(q.category);
     progressText.textContent = `Kysymys ${currentIndex + 1} / ${questions.length}`;
     questionText.textContent = q.text;
     optionsContainer.innerHTML = "";
     nextBtn.disabled = true;
     q.options.forEach((opt, index) => {
       const btn = document.createElement("button");
       btn.className = "option-btn";
       btn.textContent = opt;
       btn.addEventListener("click", () => handleAnswer(index));
       optionsContainer.appendChild(btn);
     });
   }
   function handleAnswer(selectedIndex) {
     const q = questions[currentIndex];
     const optionButtons = Array.from(document.querySelectorAll(".option-btn"));
     optionButtons.forEach((btn, index) => {
       btn.classList.add("disabled");
       btn.disabled = true;
       if (index === q.correctIndex) {
         btn.classList.add("correct");
       }
       if (index === selectedIndex && selectedIndex !== q.correctIndex) {
         btn.classList.add("wrong");
       }
     });
     if (selectedIndex === q.correctIndex) {
       totalScore++;
       categoryScores[q.category]++;
       scoreTotalEl.textContent = totalScore;
     }
     nextBtn.disabled = false;
   }
   function showResults() {
     showScreen(resultScreen);
     const social = categoryScores.social;
     const arts = categoryScores.arts;
     const math = categoryScores.math;
     socialScoreEl.textContent = `${social} / ${categoryTotals.social}`;
     artsScoreEl.textContent = `${arts} / ${categoryTotals.arts}`;
     mathScoreEl.textContent = `${math} / ${categoryTotals.math}`;
     socialCommentEl.textContent =
       social === categoryTotals.social
         ? "Erittäin vahvat sosiaaliset taidot – kuuntelet ja huomioit muita hienosti."
         : social >= 2
         ? "Hyvät sosiaaliset taidot, mutta voit vielä harjoitella ristiriitojen ratkaisua."
         : "Voit kehittää sosiaalisia taitoja harjoittelemalla kuuntelemista ja empatiaa.";
     artsCommentEl.textContent =
       arts === categoryTotals.arts
         ? "Luova ja visuaalisesti herkkä – ymmärrät värejä ja rytmiä hyvin."
         : arts >= 2
         ? "Sinulla on hyvä taidesilmä, jatka kokeilemista ja harjoittelua."
         : "Taiteelliset taidot kehittyvät tekemällä – piirrä, sävellä ja kokeile rohkeasti.";
     mathCommentEl.textContent =
       math === categoryTotals.math
         ? "Vahvat matemaattiset taidot – peruslasku ja murtoluvut hallussa."
         : math >= 2
         ? "Hyvä matikkapohja, pienellä harjoittelulla taidot vahvistuvat entisestään."
         : "Matematiikka paranee harjoittelemalla askel kerrallaan – älä luovuta.";
     overallScoreEl.textContent = `Kokonaispisteet: ${totalScore} / ${questions.length}`;
   }
   function resetQuiz() {
     currentIndex = 0;
     totalScore = 0;
categoryScores.social = 0;
     categoryScores.arts = 0;
     categoryScores.math = 0;
     scoreTotalEl.textContent = "0";
   }
   // -----------------------------
   // EVENT LISTENERS
   // -----------------------------
   startBtn.addEventListener("click", () => {
     resetQuiz();
     showScreen(quizScreen);
     renderQuestion();
   });
   nextBtn.addEventListener("click", () => {
     currentIndex++;
     if (currentIndex >= questions.length) {
       showResults();
     } else {
       renderQuestion();
     }
   });
   restartBtn.addEventListener("click", () => {
     resetQuiz();
     showScreen(quizScreen);
     renderQuestion();
   });
</script>
